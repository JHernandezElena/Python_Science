def parsear_html(texto):
    soup = BeautifulSoup(texto,'lxml')
    item = dict()
    item['titulo'] = soup.find("h1").find(text=True).replace(u'\xa0',' ').strip()
    item['ratingvalue'] = float(soup.select_one('[itemprop="ratingValue"]').text)
    return item
